

### Homeassistant

sudo docker run -d \
--name homeassistant \
--link mosquitto:mosquitto \
-p 8123:8123 \
-v /volume1/WD1/home/homeassistant:/config \
homeassistant/home-assistant

### zigbee2mqtt

sudo docker run -d \
--name zigbee2mqtt \
--device=/dev/ttyACM0 \
--link mosquitto:mosquitto \
-v /volume1/WD1/home/zigbee2mqtt:/app/data \
-v /run/udev:/run/udev:ro \
-e TZ=Europe/Berlin \
-p 8050:8050 \
koenkk/zigbee2mqtt
